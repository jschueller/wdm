134  inline double compute_test_stat(double estimate,
135  std::string method,
136  double n_eff)
137  {
138  // prevent overflow in atanh
139  if (estimate == 1.0)
140  estimate = 1 - 1e-12;
141  if (estimate == -1.0)
142  estimate = 1e-12;
143 
144  double stat;
145  if (methods::is_hoeffding(method)) {
146  stat = estimate / 30.0 + 1.0 / (36.0 * n_eff);
147  } else if (methods::is_kendall(method)) {
148  stat = estimate * std::sqrt(9 * n_eff / 4);
149  } else if (methods::is_pearson(method)) {
150  stat = boost::math::atanh(estimate) * std::sqrt(n_eff - 3);
151  } else if (methods::is_spearman(method)) {
152  stat = boost::math::atanh(estimate) * std::sqrt((n_eff - 3) / 1.06);
153  } else if (methods::is_blomqvist(method)) {
154  stat = estimate * std::sqrt(n_eff);
155  } else {
156  throw std::runtime_error("method not implemented.");
157  }
158 
159  return stat;
160  }

162  inline double compute_p_value(double statistic,
163  std::string method,
164  std::string alternative,
165  double n_eff = 0.0)
166  {
167  double p_value;
168  if (method == "hoeffd") {
169  if (n_eff == 0.0)
170  throw std::runtime_error("must provide n_eff for method 'hoeffd'.");
171  if (alternative != "two-sided")
172  throw std::runtime_error("only two-sided test available for Hoeffding's D.");
173  p_value = phoeffb(statistic, n_eff);
174  } else {
175  boost::math::normal norm_dist(0, 1);
176  if (alternative == "two-sided") {
177  p_value = 2 * boost::math::cdf(norm_dist, -std::abs(statistic));
178  } else if (alternative == "less") {
179  p_value = boost::math::cdf(norm_dist, statistic);
180  } else if (alternative == "greater") {
181  p_value = 1 - boost::math::cdf(norm_dist, statistic);
182  } else {
183  throw std::runtime_error("alternative not implemented.");
184  }
185  }
186 
187  return p_value;
188  }
