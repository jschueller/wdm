24 inline double linear_interp(const double& x,
25  const std::vector<double>& grid,
26  const std::vector<double>& values)
27 {
28  // find upper end point of interval
29  size_t i = 1;
30  while(x > grid[i]) i++;
31 
32  // linear interpolation
33  double w = (x - grid[i - 1]) / (grid[i] - grid[i - 1]);
34  return w * values[i - 1] + (1 - w) * values[i];
35 }

98 inline double calculate_test_stat(
99  const std::vector<double>& x,
100  const std::vector<double>& y,
101  std::string method,
102  std::vector<double> weights = std::vector<double>())
103 {
104  // determine effective sample size
105  double n_eff;
106  if (weights.size() == 0) {
107  n_eff = static_cast<double>(x.size());
108  } else {
109  n_eff = wdm_utils::effective_sample_size(weights);
110  }
111 
112  // calculate test statistic
113  using namespace wdm_methods;
114  double stat;
115  if (is_hoeffding(method)) {
116  stat = hoeffd(x, y, weights) / 30.0 + 1.0 / (36.0 * n_eff);
117  } else if (is_kendall(method)) {
118  stat = ktau(x, y, weights);
119  stat *= std::sqrt(9 * n_eff / 4);
120  } else if (is_pearson(method)) {
121  stat = boost::math::atanh(prho(x, y, weights));
122  stat *= std::sqrt(n_eff - 3);
123  } else if (is_spearman(method)) {
124  stat = boost::math::atanh(srho(x, y, weights));
125  stat *= std::sqrt((n_eff - 3) / 1.06);
126  } else if (is_blomqvist(method)) {
127  stat = bbeta(x, y, weights);
128  stat *= std::sqrt(n_eff);
129  } else {
130  throw std::runtime_error("method not implemented.");
131  }
132 
133  return stat;
134 }

148 inline double calculate_asymptotic_p_val(double stat,
149  std::string method,
150  double n_eff = 0.0)
151 {
152  double p_val;
153  if (method == "hoeffd") {
154  if (n_eff == 0.0)
155  throw std::runtime_error("must provide n_eff for method 'hoeffd'.");
156  p_val = phoeffb(stat, n_eff);
157  } else {
158  boost::math::normal norm_dist(0, 1);
159  p_val = 2 * boost::math::cdf(norm_dist, -std::abs(stat));
160  }
161 
162  return p_val;
163 }
